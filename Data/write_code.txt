```html

<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Chatbot</title>
    <link rel="stylesheet" href="style.css">
</head>
<body>
    <div class="chat-container">
        <div class="chat-header">
            <h2>Chatbot</h2>
        </div>
        <div class="chat-body">
            <div class="message-container">
                <p class="bot-message">Hello! How can I assist you today?</p>
            </div>
            <input type="text" id="user-input" placeholder="Type a message...">
            <button id="send-btn">Send</button>
        </div>
    </div>

    <script src="script.js"></script>
</body>
</html>
```

**JavaScript (in script.js file):**
```javascript
const chatBody = document.querySelector('.chat-body');
const userInput = document.querySelector('#user-input');
const sendBtn = document.querySelector('#send-btn');

sendBtn.addEventListener('click', () => {
    const userInputValue = userInput.value.trim();
    if (userInputValue !== '') {
        generateUserMessage(userInputValue);
        generateBotResponse(userInputValue);
        userInput.value = '';
    }
});

function generateUserMessage(message) {
    const userMessageContainer = document.createElement('div');
    userMessageContainer.className = 'message-container';
    const userMessage = document.createElement('p');
    userMessage.className = 'user-message';
    userMessage.textContent = message;
    userMessageContainer.appendChild(userMessage);
    chatBody.appendChild(userMessageContainer);
}

function generateBotResponse(message) {
    const botResponse = getBotResponse(message);
    const botMessageContainer = document.createElement('div');
    botMessageContainer.className = 'message-container';
    const botMessage = document.createElement('p');
    botMessage.className = 'bot-message';
    botMessage.textContent = botResponse;
    botMessageContainer.appendChild(botMessage);
    chatBody.appendChild(botMessageContainer);
}

function getBotResponse(message) {
    const responses = {
        'hello': 'Hi! How can I assist you today?',
        'hi': 'Hey! What\'s on your mind?',
        'how are you': 'I\'m doing great, thanks! How about you?',
        'default': 'I didn\'t quite understand that. Can you please rephrase?'
    };

    const lowerCaseMessage = message.toLowerCase();
    if (responses[lowerCaseMessage]) {
        return responses[lowerCaseMessage];
    } else {
        return responses['default'];
    }
}
```
This code creates a simple chatbot that responds to basic user inputs. You can add more functionality and responses to the chatbot by modifying the JavaScript code.
